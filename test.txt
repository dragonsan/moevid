<?php
require('simple_html_dom.php');
$mai = 'https://twist.moe/';
$output = shell_exec('curl -s "'.$mai.'"');
$main = str_get_html($output);
$main = $main->find('ul.all-anime',0)->find('li');
foreach ($main as $anime) {
    $title = trim($anime->find('span',0)->plaintext);
    $url = 'https://twist.moe'.$anime->find('a',0)->href;
    $out = shell_exec('curl -Ls "'.$url.'"');
    $html = str_get_html($out);
    $ep = $html->find('div.episode-list',0)->find('li');
    foreach ($ep as $episode) {
        $navurl = 'https://twist.moe'.$episode->find('a',0)->href;
        $outnav = shell_exec('curl -Ls "'.$navurl.'"');
        $parsenav = str_get_html($outnav);
        $parsenav = 'https://twist.moe'.$parsenav->find('video',0)->src;
        exec('curl -go test.mp4 "'.$parsenav.'"');
    }
}
